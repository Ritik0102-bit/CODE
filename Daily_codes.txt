#include<iostream>

using namespace std;

bool is_anagram(int n, int m){
    // frequency array
    int freq[9]={0};

    while(n!=0){
        freq[n%10]++;
        n/=10;
    }
    
    while(m!=0){
        freq[m%10]--;
        m/=10;
    }

    for(int i=0;i<9;i++){
        if(freq[i]!=0){
            return false;
        }
    }

    return true;
}


int main(){
    int n,m;

    n=123;
    m=3210;

    cout << is_anagram(n,m);

    
    return 0;
}

// In this code we are trying to calculate no. of possible paths from one position to the last position in a GRID

// let 3x3 GRID
// then we need to find no. of possible paths from a00 to a22
// And we can only move in right or downward direction

// Time Complexity : O(n!)
// This is a worst Time Complexity

#include<iostream>

using namespace std;

int Grid_path(int m,int n){
    if(m==0 && n==0){
        return 1;
    }
    if(m<0 || n<0){
        return 0;
    }
    return Grid_path(m-1,n) + Grid_path(m,n-1);
}

int main(){
    cout << Grid_path(3-1,3-1);

    return 0;
}

#include<iostream>

using namespace std;

void sort(int arr[],int n){
    for(int i=1;i<n;i++){
        int key=arr[i];
        int j=i-1;

        while(j>=0 && arr[j]>key){
            arr[j+1]=arr[j];
            j--;
        }
        arr[j+1]=key;
    }
}


int main(){
    
    int n=8;

    int arr[]={2,4,8,9,6,25,16,7};

    sort(arr,n);

    for(int i=0;i<n;i++){
        cout << arr[i] << "  ";
    }

    return 0;
}

we are given with a singly linked list , In which we don't have it's head , we have a ptr which is pointing to some node of that linked list & we need to delete that particular element

1 -> 2 -> 3 -> 4 -> 5 -> 6
we have ptr pointing at 4 & we need to delete that element
after deletion => 1 -> 2 -> 3 -> 5 -> 6

#include<iostream>

using namespace std;

class Node{
    public:
    int val;
    Node* next;

    Node(int data){
        val=data;
        next=NULL;
    }
};

void insert_at_tail(Node* &head,int value){
    Node* new_node=new Node(value);

    if(head==NULL){
        head=new_node;
        return;
    }

    Node* temp=head;
    while(temp->next!=NULL){
        temp=temp->next;
    }
    temp->next=new_node;
}

void print_linked_list(Node* head){
    Node* temp=head;
    while(temp!=NULL){
        cout << temp->val << " -> ";
        temp=temp->next;
    }
    cout << "NULL" << endl;
}

int main(){
    Node* linked_list=new Node(1);

    insert_at_tail(linked_list,2);
    insert_at_tail(linked_list,3);
    insert_at_tail(linked_list,4);
    insert_at_tail(linked_list,5);
    insert_at_tail(linked_list,6);

    print_linked_list(linked_list);

    Node* ptr=linked_list->next->next->next;

    ptr->val=ptr->next->val;
    
    ptr->next=ptr->next->next;

    print_linked_list(linked_list);
}

#include<iostream>

using namespace std;

class Node_2D {
    public:
    int val;
    Node_2D* next;
    Node_2D* down;

    Node_2D(int data){
        val=data;
        next=NULL;
        down=NULL;
    }
};

void insert_in_2D_LL(Node_2D* &head,int row,int col,int value){
    Node_2D* new_node=new Node_2D(value);

    if(head==NULL){
        head=new_node;
        return;
    }

    Node_2D* temp=head;
    for(int i=0;i<row;i++){
        temp=temp->down;
    }

    for(int i=0;i<col;i++){
        temp=temp->next;
    }
    temp=new_node;
}

void insert_in_2D_LL_col(Node_2D* &head,int value){
    Node_2D* new_node=new Node_2D(value);
    
    if(head==NULL){

        return;
    }
    
    Node_2D* temp=head;
    while(temp->down!=NULL){
        temp=temp->down;
    }
    temp->down=new_node;
}

#include <stdio.h>
#include <stdlib.h>

struct ColNode {
    int data;
    struct ColNode *right;
};

struct RowNode {
    // int data;
    struct RowNode *down;
    struct ColNode *right;   
};


void build(struct RowNode **head) {
    int input = 0;
    int rows = 4;
    int cols = 4;

    *head = NULL;
    struct RowNode *lastRow = NULL;

    for (int i = 0; i < rows; ++i) {
      
        struct RowNode *newRow = (struct RowNode *)malloc(sizeof(struct RowNode));
      
        
        newRow->down = NULL;
        newRow->right = NULL;

       
        if (lastRow == NULL) {
            *head = newRow;
            lastRow = newRow;
        } else {
            lastRow->down = newRow;
            lastRow = newRow;
        }

        
        struct ColNode *lastCol = NULL;
        for (int j = 0; j < cols; ++j) {
            struct ColNode *newCol = (struct ColNode *)malloc(sizeof(struct ColNode));
        
            newCol->data = input++;
            newCol->right = NULL;

            if (lastCol == NULL) {
               
                newRow->right = newCol;
                lastCol = newCol;
            } else {
                lastCol->right = newCol;
                lastCol = newCol;
            }
        }
    }
}


void printGrid(struct RowNode *head) {
    struct RowNode *r = head;
    int rowIndex = 0;
    while (r) {
        printf("Row: ");

        struct ColNode *c = r->right;
        while (c) {
            printf("%d -> ", c->data);
            c = c->right;
        }
        printf("NULL\n");
        r = r->down;
    }
}



int main(void) {
    struct RowNode *grid = NULL;
    build(&grid);

    printGrid(grid);


    return 0;
}
